(function () {
 $(".mainNavCon").outerHeight($(window).height()-$(".headerCon").outerHeight()-$(".footerCon").outerHeight());
 $(".subNavList li").height($(".mainNavCon").height()-parseInt($(".subNavList li").css('padding-top'))*2);
 $('.mainNav').click(function () {
 if ($('.mainNavCon').css('display') == 'none') {
 $('.mainNavCon').fadeIn();
 }
 else {
 $('.mainNavCon').fadeOut();
 var index=$(".treeNav a.on").parent().parent().parent().index();
 $(".subNavList li").eq(index).fadeIn();
 $(".mainNavList a").eq(index+1).addClass('on').siblings().removeClass('on');
 }
 $(".treeNav").fadeOut();
 $(".treeNav a.on").parent().addClass('on');
 });
 $('.mainNavConBg').click(function () {
 $('.mainNavCon').fadeOut();
 });
 $('.mainNavList a').click(function () {
 var num = $(this).index();
 $(this).addClass('on').siblings().removeClass('on');
 $('.subNavList li').eq(num - 1).stop(true, true).fadeIn().siblings().stop(true, true).fadeOut();
 });
 $(".subNavList li .subNavCon > a").click(function () {
 if($(this).siblings().length!==0){
 $(".subNavList li .subNavCon > a").removeClass('on');
 $(this).parent().siblings().find('.treeNav').fadeOut();
 $(this).siblings().slideToggle();
 $(this).addClass('on');
 }
 else {
 $(".treeNav a").removeClass('on');
 $(".subNavList li .subNavCon > .treeNav").slideUp();
 $('.mainNavCon').fadeOut();
 }
 });
 $(".treeNav a").click(function () {
 $(".treeNav a").removeClass('on');
 $(this).addClass('on');
 $(".mainNavCon").fadeOut();
 $(".treeNav").slideToggle();
 });
})();
